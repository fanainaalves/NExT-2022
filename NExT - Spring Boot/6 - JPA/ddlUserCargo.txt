CREATE TABLE CAR_CARGO (
     CAR_ID MEDIUMINT NOT NULL AUTO_INCREMENT,
     CAR_NOME VARCHAR(60) NOT NULL,
     PRIMARY KEY (CAR_ID)
);

CREATE TABLE USR_USER (
     USR_ID MEDIUMINT NOT NULL AUTO_INCREMENT,
     USR_CPF CHAR(11) NOT NULL,
     USR_NOME VARCHAR(60) NOT NULL,
     CAR_ID MEDIUMINT,
     PRIMARY KEY (USR_ID), 
     FOREIGN KEY(CAR_ID) REFERENCES CAR_CARGO(CAR_ID)
);
